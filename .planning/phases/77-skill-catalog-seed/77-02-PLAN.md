```python
import os, yaml
skills_dir = "skills"
target_20 = {"flywheel","meeting-prep","meeting-processor","call-intelligence",
    "account-research","sales-collateral","outreach-drafter","gtm-pipeline",
    "gtm-my-company","gtm-company-fit-analyzer","gtm-web-scraper-extractor",
    "gtm-outbound-messenger","legal","investor-update","brainstorm","spec",
    "pricing","social-media-manager","demo-prep","demo"}
enabled_count = 0
for d in sorted(os.listdir(skills_dir)):
    p = os.path.join(skills_dir, d, "SKILL.md")
    if not os.path.isfile(p): continue
    fm = yaml.safe_load(open(p).read().split("---")[1])
    if not fm: continue
    is_enabled = fm.get("enabled", True)
    has_triggers = bool(fm.get("triggers"))
    has_tags = bool(fm.get("tags"))
    name = fm.get("name", d)
    if is_enabled:
        enabled_count += 1
        assert name in target_20, f"{name} enabled but not in target 20"
        assert has_triggers, f"{name} missing triggers"
        assert has_tags, f"{name} missing tags"
        print(f"  ENABLED: {name} triggers={len(fm['triggers'])} tags={fm['tags']}")
    else:
        assert name not in target_20, f"{name} disabled but IS in target 20"
assert enabled_count == 20, f"Expected 20 enabled, got {enabled_count}"
print(f"\nPASS: {enabled_count} enabled skills, all with triggers and tags")
```
