Build 4 task types as separate queries, each returning project-level tasks:

**1. "review" tasks (priority 1 — highest):**
- Query: `BrokerProject` where `status='gaps_identified'`, `approval_status='draft'`, tenant scoped, not deleted
- Sort: `created_at ASC` (oldest first = most urgent)
- Each row becomes: `{"type": "review", "priority": 1, "project_id": str(id), "project_name": name, "created_at": iso, "message": "Review extracted coverages"}`

**2. "approve" tasks (priority 2):**
- Query: `BrokerProject` where `status='gaps_identified'`, `approval_status='approved'`, AND has at least one `CarrierQuote` with `draft_status='pending'`. Use `exists()` subquery pattern.
- Sort: `created_at ASC`
- Each row becomes: `{"type": "approve", "priority": 2, "project_id": str(id), "project_name": name, "created_at": iso, "message": "Approve & send solicitations"}`

**3. "export" tasks (priority 3):**
- Query: `BrokerProject` where `status IN ('quotes_complete', 'recommended')`, tenant scoped, not deleted
- Sort: `created_at ASC`
- Each row becomes: `{"type": "export", "priority": 3, "project_id": str(id), "project_name": name, "created_at": iso, "message": "Export comparison for client"}`

**4. "followup" tasks (priority 4):**
- Query: `CarrierQuote` joined to `BrokerProject`, where `CarrierQuote.draft_status='sent'` AND `CarrierQuote.solicited_at` is older than 7 days AND `CarrierQuote.status='pending'` (no quotes received yet). Group by project.
- Sort: `solicited_at ASC` (most overdue first)
- Use `func.now() - CarrierQuote.solicited_at > text("interval '7 days'")` for the overdue check
- Each row becomes: `{"type": "followup", "priority": 4, "project_id": str(id), "project_name": name, "solicited_at": iso, "carrier_name": carrier_config.carrier_name, "days_overdue": N, "message": "Follow up with {carrier_name}"}`
- NOTE: followup tasks are per-carrier-quote, not per-project. Include `quote_id` in the task object.

Combine all 4 lists into a single array, already ordered by priority then created_at within each group. Do NOT re-sort — just concatenate in priority order (review + approve + export + followup).

Return: `{"tasks": [...], "total": N}`

   ```python
   def _build_comparison_xlsx(comparison: dict, project_name: str) -> bytes:
       import openpyxl
       from openpyxl.styles import Font, PatternFill, Alignment, Border, Side
       
       wb = openpyxl.Workbook()
       # Remove default sheet
       wb.remove(wb.active)
       
       # Group coverages by category
       categories = {}
       for coverage_id, cov_data in comparison.get("coverages", {}).items():
           cat = cov_data.get("category", "insurance")
           categories.setdefault(cat, []).append(cov_data)
       
       # Create one sheet per category (Insurance, Surety)
       for cat_name, cov_list in categories.items():
           ws = wb.create_sheet(title=cat_name.capitalize())
           # Row 1: Headers — Coverage Type | Carrier 1 | Carrier 2 | ...
           # Collect unique carriers from quotes across all coverages
           carrier_names = []
           carrier_map = {}  # carrier_id -> name
           for cov in cov_list:
               for quote in cov.get("quotes", []):
                   cid = quote.get("carrier_config_id") or quote.get("carrier_id")
                   if cid and cid not in carrier_map:
                       carrier_map[cid] = quote.get("carrier_name", "Unknown")
                       carrier_names.append(quote.get("carrier_name", "Unknown"))
           
           # Write header row
           headers = ["Coverage", "Required Limit"] + carrier_names
           for col, h in enumerate(headers, 1):
               cell = ws.cell(row=1, column=col, value=h)
               cell.font = Font(bold=True)
               cell.fill = PatternFill(start_color="4472C4", end_color="4472C4", fill_type="solid")
               cell.font = Font(bold=True, color="FFFFFF")
           
           # Write coverage rows
           row = 2
           for cov in cov_list:
               ws.cell(row=row, column=1, value=cov.get("display_name", cov.get("coverage_type", "")))
               ws.cell(row=row, column=2, value=cov.get("required_limit"))
               
               # Fill in each carrier's quote for this coverage
               for q in cov.get("quotes", []):
                   cid = q.get("carrier_config_id") or q.get("carrier_id")
                   if cid in carrier_map:
                       col_idx = list(carrier_map.keys()).index(cid) + 3  # offset by 2 fixed cols
                       premium = q.get("premium") or q.get("quoted_premium")
                       ws.cell(row=row, column=col_idx, value=premium)
                       
                       # Red fill for EXCLUDED
                       if q.get("has_exclusion"):
                           ws.cell(row=row, column=col_idx).fill = PatternFill(
                               start_color="FFC7CE", end_color="FFC7CE", fill_type="solid"
                           )
               row += 1
           
           # Auto-width columns
           for col in ws.columns:
               max_len = max((len(str(cell.value or "")) for cell in col), default=10)
               ws.column_dimensions[col[0].column_letter].width = min(max_len + 2, 40)
       
       # If no categories found, create a single empty sheet
       if not categories:
           wb.create_sheet(title="No Data")
       
       buf = io.BytesIO()
       wb.save(buf)
       return buf.getvalue()
   ```
