```typescript
export interface PipelineListItem {
  id: string
  name: string
  domain: string | null
  entity_type: 'company' | 'person'
  stage: string
  fit_score: number | null
  fit_tier: string | null
  relationship_type: string[]
  source: string
  channels: string[]
  ai_summary: string | null
  last_activity_at: string | null
  created_at: string | null
  contact_count: number
  primary_contact: { name: string; email: string | null; title: string | null } | null
}

export interface PipelineResponse {
  items: PipelineListItem[]
  total: number
  offset: number
  limit: number
}

export type ViewTab = 'all' | 'needs_action' | 'replied' | 'stale'

export interface PipelineParams {
  offset?: number
  limit?: number
  search?: string
  sort_by?: string
  sort_dir?: 'asc' | 'desc'
  stage?: string[]
  fit_tier?: string[]
  relationship_type?: string[]
  source?: string
  view?: ViewTab
}

export type EditableField = 'stage' | 'fit_tier'
```

**Rewrite `api.ts`** — Remove `fetchPipelineIndustries`, `GraduatePayload`, `graduateAccount`. Replace with:
- `fetchPipeline(params: PipelineParams): Promise<PipelineResponse>` — GET /pipeline/ with params serialized (arrays as comma-separated). Keep existing serialization pattern for array params.
- `patchPipelineEntry(id: string, data: Record<string, unknown>): Promise<PipelineListItem>` — PATCH /pipeline/{id}
- `createPipelineEntry(data: { name: string; entity_type?: string; domain?: string }): Promise<{ entry: PipelineListItem; dedup_matched: boolean; status: number }>` — POST /pipeline/, capture response status to detect dedup (200 vs 201). Parse the response body and attach the status code.
- `searchPipeline(q: string): Promise<PipelineListItem[]>` — GET /pipeline/search?q=...

**Rewrite `hooks/usePipeline.ts`** — TanStack Query hook:
```typescript
export function usePipeline(params: PipelineParams) {
  return useQuery({
    queryKey: ['pipeline', params],
    queryFn: () => fetchPipeline(params),
    staleTime: 30_000,
    placeholderData: keepPreviousData,
  })
}
```
Import `keepPreviousData` from `@tanstack/react-query` to prevent flash-of-empty on page/filter changes.

**Create `hooks/usePipelineMutation.ts`** — Optimistic update mutation following research Pattern 2:
- `usePipelineMutation()` returns a mutation that calls `patchPipelineEntry`
- On mutate: cancel outgoing pipeline queries, snapshot cache, optimistically update the matching item in all cached pipeline query results
- On error: revert to snapshot, show `toast.error('Failed to update')`
- On settled: invalidate all pipeline queries
- Import `toast` from `sonner`, `useQueryClient` from `@tanstack/react-query`

**Create `hooks/usePipelineCreate.ts`** — Create mutation:
- `usePipelineCreate()` returns a mutation calling `createPipelineEntry`
- On success: if `dedup_matched` is true, show `toast.info('Found existing entry: ${name}')`. If false, show `toast.success('Created: ${name}')`.
- On error: `toast.error('Failed to create entry')`
- On settled: invalidate pipeline queries
